8.def number(str):
	num1 = 0
	num2 = 0
	num3 = 0
	num4 = 0
	for a in str:
		if a.isdigit() == True:
			num1 += 1
		elif a.isalpha() == True:
			num2 += 1
		elif a.isspace() == True:
			num3 += 1
		else:
			num4 +=1
	print("数字有%d个"%num1)
	print("字母有%d个"%num2)
	print("空格有%d个"%num3)
	print("其它有%d个"%num4)
